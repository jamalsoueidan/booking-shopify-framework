description: A CreateManyShifts component.
---

This component a shift and has a onSubmit handler that sends an event when the form is submitted successfully.
You must handle submit via ref because this component can be used in both modal and other pages that need to put the submit button in different places on the page (see example).

```json
[
  {
    "days": ["thursday"],
    "start": "2023-01-23T08:00:00.000Z",
    "end": "2023-01-23T15:00:00.000Z",
    "tag": "#4b6043"
  }
]
```

### Component usage

```tsx
const ref = useRef < CreateManyShiftsRefMethod > null;
const onSubmit = useCallback((fieldValues): CreateManyShiftsSubmitResult => {
  return {
    status: "success",
  };
}, []);

const initData: ScheduleFormManyShiftsBody = {
  days: ["monday"],
  end: addMonths(setHours(new Date(), 10), 1),
  start: setHours(new Date(), 16),
  tag: Tag.all_day,
};

const submit = useCallback(() => {
  ref?.current?.submit();
}, [ref]);

return (
  <>
    <ScheduleFormManyShifts data={initData} onSubmit={onSubmit} ref={ref} />
    <Button onClick={submit}>Submit</Button>
  </>
);
```
