description: A CreateOneShift component.
---

This component generates one shift and has a onSubmit handler that dispatches an event when the form is successfully submitted.
You must handle submit via ref, as this component can be used in modal and other pages, that needs to place submit button different locations on the page (see example)!

### Component usage

```js
const ref = useRef<CreateOneShiftRefMethod>();

const onSubmit = useCallback((fieldValues): SubmitResult => {
  return {
    status: "success",
  };
}, []);

const submit = useCallback(() => {
  ref?.current?.submit();
}, [ref]);

return (
  <Card
    sectioned
    primaryFooterAction={{ content: "Submit", onAction: submit }}
  >
    <CreateOneShift
      selectedDate={new Date().toDateString()}
      onSubmit={onSubmit}
      ref={ref}
    />
  </Card>
);
```
